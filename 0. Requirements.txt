mport numpy as np
import matplotlib.pyplot as plt
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.linalg import eigsh
plt.style.use("seaborn-v0_8")

eps = 1e-12

def make_points(N, dim=4, seed=0):
rng = np.random.default_rng(seed)
X = rng.normal(size=(N,dim))
X /= np.linalg.norm(X,axis=1,keepdims=True)
return X

def make_rho_g(X, beta=1.0):
D = squareform(pdist(X)).astype(np.float32)
rho = np.exp(-D/beta)
rho /= rho.max() + eps
g = -np.log(rho + eps)
return rho, g

def fig_spectrum_log512():
X = make_points(512)
_, g = make_rho_g(X)
vals = eigsh(g,k=40,which='LA',return_eigenvectors=False)
vals = np.sort(np.abs(vals))[::-1]
plt.semilogy(vals,'o-')
plt.savefig("paper/figures/fig_spectrum_log512.png")

if name == "main":
fig_spectrum_log512()
